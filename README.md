            const cpuFreq = parseInt(document.getElementById('cpuSlider').value);
            const gpuFreq = parseInt(document.getElementById('gpuSlider').value);
            const ramFreq = parseInt(document.getElementById('ramSlider').value);
            const game = document.getElementById('gameSelect').value;
            const profile = document.getElementById('profileSelect').value;

            // Performance calculation (simplified model)
            const cpuScore = (cpuFreq / 1785) * 40;
            const gpuScore = (gpuFreq / 1267) * 40;
            const ramScore = (ramFreq / 1600) * 20;
            const performance = Math.min(100, cpuScore + gpuScore + ramScore);

            // Temperature calculation (simplified model)
            const baseTemp = 40;
            const cpuTemp = baseTemp + (cpuFreq / 1785) * 35;
            const gpuTemp = baseTemp + (gpuFreq / 1267) * 40;
            const avgTemp = (cpuTemp + gpuTemp) / 2;

            // Power consumption calculation (simplified model)
            const cpuPower = (cpuFreq / 1785) * 8;
            const gpuPower = (gpuFreq / 1267) * 7;
            const ramPower = (ramFreq / 1600) * 2;
            const basePower = 3;
            const totalPower = basePower + cpuPower + gpuPower + ramPower;

            // Stability calculation (inverse of temperature and power)
            const stability = Math.max(20, 100 - (avgTemp - 40) - (totalPower - 5) * 2);

            return {
                performance,
                stability,
                cpuTemp,
                gpuTemp,
                avgTemp,
                totalPower,
                componentUsage: [
                    Math.min(100, cpuScore * 1.5),
                    Math.min(100, gpuScore * 1.5),
                    Math.min(100, ramScore * 2),
                    Math.min(100, performance * 0.5),
                    Math.min(100, performance * 0.4)
                ]
            };
        }

        // Update UI with calculated metrics
        function updateUI(metrics) {
            try {
                // Update status cards
                document.getElementById('performanceScore').textContent = `${Math.round(metrics.performance)}%`;
                document.getElementById('stabilityScore').textContent = `${Math.round(metrics.stability)}%`;
                document.getElementById('tempValue').textContent = `${Math.round(metrics.avgTemp)}¬∞C`;
                document.getElementById('powerValue').textContent = `${metrics.totalPower.toFixed(1)}W`;

                // Update temperature status indicator
                const tempStatus = document.getElementById('tempStatus');
                tempStatus.className = 'status-indicator ';
                if (metrics.avgTemp < 60) {
                    tempStatus.className += 'status-good';
                } else if (metrics.avgTemp < 80) {
                    tempStatus.className += 'status-warning';
                } else {
                    tempStatus.className += 'status-critical';
                }

                // Update component chart
                if (componentChart) {
                    componentChart.data.datasets[0].data = metrics.componentUsage;
                    componentChart.update();
                }
            } catch (error) {
                console.error('Error updating UI:', error);
            }
        }

        // Simulate system behavior over time
        function simulateTimeStep() {
            const metrics = calculateSystemMetrics();

            // Add some randomness to simulate real-world fluctuations
            const perfWithNoise = metrics.performance + (Math.random() - 0.5) * 5;
            const stabilityWithNoise = metrics.stability + (Math.random() - 0.5) * 3;
            const cpuTempWithNoise = metrics.cpuTemp + (Math.random() - 0.5) * 2;
            const gpuTempWithNoise = metrics.gpuTemp + (Math.random() - 0.5) * 2;
            const powerWithNoise = metrics.totalPower + (Math.random() - 0.5) * 0.5;

            // Update time labels
            const now = new Date();
            const timeString = `${now.getMinutes()}:${now.getSeconds().toString().padStart(2, '0')}`;
            timeLabels.push(timeString);
            
            // Keep only last 20 data points
            if (timeLabels.length > 20) {
                timeLabels.shift();
                performanceData.shift();
                powerData.shift();
                performanceChart.data.datasets[0].data.shift();
                performanceChart.data.datasets[1].data.shift();
                temperatureChart.data.datasets[0].data.shift();
                temperatureChart.data.datasets[1].data.shift();
            }

            // Update data arrays
            performanceData.push(Math.max(0, Math.min(100, perfWithNoise)));
            powerData.push(Math.max(0, powerWithNoise));
            performanceChart.data.datasets[0].data.push(Math.max(0, Math.min(100, perfWithNoise)));
            performanceChart.data.datasets[1].data.push(Math.max(0, Math.min(100, stabilityWithNoise)));
            temperatureChart.data.datasets[0].data.push(Math.max(30, cpuTempWithNoise));
            temperatureChart.data.datasets[1].data.push(Math.max(30, gpuTempWithNoise));

            // Update charts
            if (performanceChart) performanceChart.update();
            if (temperatureChart) temperatureChart.update();
            if (powerChart) powerChart.update();

            // Update UI
            updateUI(metrics);

            // Update learning progress based on current settings
            updateLearningProgress();
        }
